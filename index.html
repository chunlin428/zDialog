点击空白处就可关闭弹出框

1、内容设置

文本内容 标签内容 
				
var options1_1 = {content:'这是弹框文本内容'};
var options1_2 = {content:'< p style="color:red">这是标签内容'};

2、位置设置

左右修改 top设置 fixed 
				
var options2_1 = {left:'20%', right:'20%'};
var options2_2 = {top:'20%'};
var options2_3 = {fixed:true};

3、按钮设置

确定按钮 取消按钮 按钮组 
				
var okFn = function(){alert('点击出现确定')};
var options3_1 = {ok:okFn, okValue:'点击出现确定', okCSS:'btn'};
				
var cancelFn = function(){dialog.close()};
var options3_2 = {cancel:cancelFn, cancelValue:'点击关闭弹框', cancelCSS:'btn btn-primary'};
				
var btns = [
	{value:'按钮1',callback:function(){alert('按钮1')},css:'btn btn-success'},
	{value:'按钮2',callback:function(){alert('按钮2')},css:'btn btn-warning'}
];
var options3_3 = {buttons:btns};

4、交互设置

快速关闭 阴影透明 点击事件 
				
var options4_1 = {quickClose:false, content:'点击外面关闭不了了'};
var options4_2 = {opacity:0.2, content:'阴影变淡了'};
var options4_3 = {eventType:'tap', content:'使用tap事件'};

5、事件

打开的时候触发 关闭的时候触发 
				
var options5_1 = {onopen:function(){alert('在打开的时候触发')}};
var options5_2 = {onclose:function(){alert('在关闭的时候触发')}};
